peoples.data <- data.frame(
   id = c (1:3), 
   name = c("Czlowiek1", "Czlowiek2", "Czlowiek3"),
   years_old = c(18, 21, 66), 
   stringsAsFactors = FALSE
)
ggplot(data = peoples.data, mapping = aes(name, years_old, fill=name)) + geom_bar(width=1, stat="identity")
